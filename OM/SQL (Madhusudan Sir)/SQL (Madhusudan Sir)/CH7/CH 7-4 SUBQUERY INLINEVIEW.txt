1 DISPLAY FIRST_NAME , DEPARTMENT_ID , SALARY , HIGHEST SALARY PAID TO THE CORRESPONDING DEPARTMENT_ID
 -> SELECT FIRST_NAME, DEPARTMENT_ID, SALARY, MAX_SALARY
	FROM EMPLOYEES
	NATURAL JOIN (SELECT DEPARTMENT_ID, MAX(SALARY) MAX_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

2 DISPLAY LAST_NAME , JOB_ID , SALARY , LOWEST SALARY PAID TO THE CORRESPONDING JOB_ID
 -> SELECT LAST_NAME, JOB_ID, SALARY, MIN_JOB_SALARY
	FROM EMPLOYEES
	NATURAL JOIN (SELECT JOB_ID, MIN(SALARY) MIN_JOB_SALARY FROM EMPLOYEES GROUP BY JOB_ID);		

3 DISPLAY FIRST_NAME , MANAGER_ID , SALARY , AVERAGE SALARY PAID TO THE CORRESPONDING MANAGER_ID
 -> SELECT FIRST_NAME, MANAGER_ID, SALARY, MGR_AVG_SALARY
	FROM EMPLOYEES
	NATURAL JOIN (SELECT MANAGER_ID, TRUNC(AVG(SALARY)) MGR_AVG_sALARY FROM EMPLOYEES GROUP BY MANAGER_ID);

4 DISPLAY FIRST_NAME , HIRE YEAR , SALARY , HIRE YEAR WISE LOWEST SALARY , HIRE YEAR WISE HIGHEST SALARY
 -> SELECT FIRST_NAME , HIRE_YEAR , SALARY , HY_MIN_SAL , HY_MAX_SAL 
	FROM (SELECT TO_CHAR(HIRE_DATE,'RRRR') HIRE_YEAR , FIRST_NAME , SALARY FROM EMPLOYEES)
	NATURAL JOIN(SELECT TO_CHAR(HIRE_DATE,'RRRR') HIRE_YEAR , MIN(SALARY) HY_MIN_SAL , MAX(SALARY) HY_MAX_SAL FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,'RRRR'));

5 DISPLAY FIRST_NAME , SALARY , DEPARTMENT_ID , DEPARTMENT WISE HIGHEST SALARY , JOB_ID , JOB WISE LOWEST SALARY
 -> SELECT FIRST_NAME, SALARY, DEPARTMENT_ID, DEPT_HIGH_SAL, JOB_ID, JOB_LOW_SAL
	FROM EMPLOYEES
	NATURAL JOIN (SELECT DEPARTMENT_ID, MAX(SALARY) DEPT_HIGH_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
	NATURAL JOIN (SELECT JOB_ID, MIN(SALARY) JOB_LOW_SAL FROM EMPLOYEES GROUP BY JOB_ID);

6 DISPLAY LAST_NAME , DEPARTMENT_ID , SALARY , LOWEST SALARY , HIGHEST SALARY FOR THE CORRESPONDING DEPARTMENT_ID
 -> SELECT LAST_NAME, DEPARTMENT_ID, SALARY, DEPT_LOW_SAL, DEPT_HIGH_SAL
	FROM EMPLOYEES 
	NATURAL JOIN (SELECT DEPARTMENT_ID, MIN(SALARY) DEPT_LOW_SAL, MAX(SALARY) DEPT_HIGH_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

7 DISPLAY FIRST_NAME , JOB_ID , SALARY , HIGHEST SALARY , DIFFERENCE (HIGHEST SALARY - SALARY) FOR THE CORRESPONDING JOB_ID
 -> SELECT FIRST_NAME, JOB_ID, SALARY, MAX_SALARY, MAX_SALARY - SALARY DIFFERENCE
	FROM EMPLOYEES
	NATURAL JOIN (SELECT JOB_ID, MAX(SALARY) MAX_SALARY FROM EMPLOYEES GROUP BY JOB_ID);

8 DISPLAY FIRST_NAME , LAST_NAME , DEPARTMENT_ID , JOB_ID , NO. OF EMPLOYEES WORKING FOR DEPARTMENT_ID , NO. OF EMPLOYEES WORKING FOR JOB_ID
 -> SELECT FIRST_NAME, LAST_NAME, DEPARTMENT_ID, JOB_ID, EMP_IN_DEPT, EMP_IN_JOB
	FROM EMPLOYEES
	NATURAL JOIN (SELECT DEPARTMENT_ID, COUNT(*) EMP_IN_DEPT FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
	NATURAL JOIN (SELECT JOB_ID, COUNT(*) EMP_IN_JOB FROM EMPLOYEES GROUP BY JOB_ID);

9 DISPLAY FIRST_NAME , JOB_ID , SALARY , TOTAL SALARY PAID TO JOB_ID , %OF SALARY EMPLOYEE GET FOR THE CORRESPONDING JOB_ID
 -> SELECT FIRST_NAME, JOB_ID, SALARY, JOB_TOTAL_SALARY, SALARY/JOB_TOTAL_SALARY "% OF EMP GET SALARY"
	FROM EMPLOYEES
	NATURAL JOIN (SELECT JOB_ID, SUM(SALARY) JOB_TOTAL_SALARY FROM EMPLOYEES GROUP BY JOB_ID);

10 DISPLAY FIRST_NAME , DEPARTMENT_ID , SALARY , LOWEST SALARY , DIFFERENCE IN % FOR THE CORRESPONDING DEPARTMENT_ID
 -> SELECT FIRST_NAME, DEPARTMENT_ID, SALARY, LOW_SAL, TRUNC(100 - (LOW_SAL * 100)/SALARY) "DIFFERENCE IN %"
	FROM EMPLOYEES
	NATURAL JOIN (SELECT DEPARTMENT_ID, MIN(SALARY) LOW_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)

11 DISPLAY FIRST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE HIGHEST SALARY , JOB_ID , JOB WISE LOWEST SALARY
 -> SELECT FIRST_NAME, SALARY, DEPARTMENT_ID, DEPT_HIGH_SAL, JOB_ID, JOB_LOW_SAL
	FROM EMPLOYEES
	NATURAL JOIN (SELECT DEPARTMENT_ID, MAX(SALARY) DEPT_HIGH_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
	NATURAL JOIN (SELECT JOB_ID, MIN(SALARY) JOB_LOW_SAL FROM EMPLOYEES GROUP BY JOB_ID);

12 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE AVERAGE SALARY , MANAGER_ID , MANAGER WISE NO. OF EMPLOYEES
 -> SELECT LAST_NAME, SALARY, DEPARTMENT_ID, DEPT_AVG_SAL, MANAGER_ID, MGR_COUNT_EMP
	FROM EMPLOYEES
	NATURAL JOIN (SELECT DEPARTMENT_ID, TRUNC(AVG(SALARY)) DEPT_AVG_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
	NATURAL JOIN (SELECT MANAGER_ID, COUNT(*) MGR_COUNT_EMP FROM EMPLOYEES GROUP BY MANAGER_ID);	

13 DISPLAY LAST_NAME , SALARY , MANAGER_ID , MGR WISE AVERAGE SALARY , JOB_ID , JOB WISE NO. OF EMPLOYEES
 -> SELECT LAST_NAME, SALARY, MANAGER_ID, MGR_AVG_SAL, JOB_ID, JOB_COUNT_EMP
	FROM EMPLOYEES
	NATURAL JOIN (SELECT MANAGER_ID, TRUNC(AVG(SALARY)) MGR_AVG_SAL FROM EMPLOYEES GROUP BY MANAGER_ID)
	NATURAL JOIN (SELECT JOB_ID, COUNT(*) JOB_COUNT_EMP FROM EMPLOYEES GROUP BY JOB_ID);

14 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE LOWEST SALARY , JOB_ID , JOB WISE AVERAGE SALARY, MANAGER_ID , MANAGER WISE HIGHEST SALARY
 -> SELECT LAST_NAME, SALARY, DEPARTMENT_ID, DEPT_MIN_SAL, JOB_ID, JOB_AVG_SAL, MANAGER_ID, MGR_MAX_SAL
	FROM EMPLOYEES
	NATURAL JOIN (SELECT DEPARTMENT_ID, MIN(SALARY) DEPT_MIN_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
	NATURAL JOIN (SELECT JOB_ID, TRUNC(AVG(SALARY)) JOB_AVG_SAL FROM EMPLOYEES GROUP BY JOB_ID)
	NATURAL JOIN (SELECT MANAGER_ID, MAX(SALARY) MGR_MAX_SAL FROM EMPLOYEES GROUP BY MANAGER_ID);

16 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE AVERAGE SALARY , MANAGER_ID , MANAGER WISE NO. OF EMPLOYEES , DEPARTMENT_NAME , CITY

18 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE LOWEST SALARY , JOB_ID , JOB WISE AVERAGE SALARY, MANAGER_ID , MANAGER WISE HIGHEST SALARY , DEPARTMENT_NAME , CITY , COUNTRY_NAME , REGION_NAME
