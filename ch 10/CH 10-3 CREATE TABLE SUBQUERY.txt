SQL> CREATE TABLE R1 AS SELECT * FROM REGIONS;

2 CREATE TABLE E2 USING EMPLOYEES TABLE WHICH WILL HAVE EMPLOYEE_ID , JOB_ID , SALARY COLUMNS ONLY.

SQL> CREATE TABLE E2 AS SELECT EMPLOYEE_ID, JOB_ID, SALARY FROM EMPLOYEES ;

3 CREATE TABLE E3 USING EMPLOYEES TABLE WHICH WILL HAVE ONLY ROWS HAVING SALARY HIGHER THAN 10000.

SQL> CREATE TABLE E3 AS SELECT * FROM EMPLOYEES WHERE SALARY > 10000;

4 CREATE TABLE E4 USING EMPLOYEES TABLE WHICH WILL HAVE FIRST_NAME , SALARY , JOB_ID , HIRE_DATE COLUMNS. INCLUDE ONLY THOSE ROWS WHICH WERE HIRED BEFORE 1995.

SQL> CREATE TABLE E4 AS SELECT FIRST_NAME, SALARY, JOB_ID, HIRE_DATE FROM EMPLOYEES WHERE TO_CHAR(HIRE_DATE,'YYYY') < 1995;

5 CREATE TABLE D5 USING DEPARTMENTS TABLE. ONLY DEFINITION SHOULD BE CREATED WITHOUT ANY DATA.

SQL> CREATE TABLE D5 AS SELECT * FROM DEPARTMENTS WHERE 1=2;

6 CREATE TABLE ED6 USING EMPLOYEES AND DEPARTMENTS TABLES. INCLUDE FIRST_NAME AND DEPARTMENT_NAME WITH ALL ROWS.

SQL> CREATE TABLE ED6 AS SELECT FIRST_NAME, DEPARTMENT_NAME FROM EMPLOYEES JOIN DEPARTMENTS USING (DEPARTMENT_ID);

7 CREATE TABLE E7 USING EMPLOYEES TABLE. INCLUDE JOB_ID AND HIGHEST SALRAY PAID TO EACH JOB_ID.

SQL> CREATE TABLE E7 AS SELECT JOB_ID, MAX(SALARY) MAX_SALARY FROM EMPLOYEES GROUP BY JOB_ID;

8 CREATE TABLE E8 USING EMPLOYEES TABLE. INCLUDE FIRST_NAME AND ANNUAL SALARY.

SQL> CREATE TABLE E8 AS SELECT FIRST_NAME, SALARY*12 ANNUAL_SAL FROM EMPLOYEES;

9 CREATE TABLE E9 USING EMPLOYEES AND DEPARTMENTS TABLES. INCLUDE ROWS WHICH WORKS IN Shiping DEPARTMENT.

SQL> CREATE TABLE E9 AS SELECT * FROM EMPLOYEES E
	JOIN DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID 
	WHERE DEPARTMENT_NAME = 'Shiping';

10 CREATE TABLE E10 USING DEPARTMENTS AND EMPLOYEES TABLES. INCLUDE DEPARTMENT_NAME , LOWEST SALARY , HIGHEST SALARY PAID TO EACH DEPARTMENT.

SQL> CREATE TABLE E10 AS SELECT DEPARTMENT_NAME, LOWEST_SAL, HIGHEST_SAL FROM DEPARTMENTS
	JOIN (SELECT DEPARTMENT_ID, MAX(SALARY) HIGHEST_SAL, MIN(SALARY) LOWEST_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID) 
	USING (DEPARTMENT_ID);


11 CREATE TABLE E11 WHICH WILL HAVE CAPITAL FIRST_NAME , LOWER EMAIL 

SQL> CREATE TABLE E11 AS SELECT UPPER(FIRST_NAME), LOWER(EMAIL) FROM EMPLOYEES;

12 CREATE TABLE E12 WHICH WILL HAVE LAST_NAME , DEPARTMENT_NAME

SQL> CREATE TABLE E12 AS SELECT LAST_NAME, DEPARTMENT_NAME FROM EMPLOYEES

13 CREATE TABLE E13 WHICH HAS COLUMN NAMES EDNAME. THIS SHOULD STORE ALL FIRST_NAME AND DEPARTMENT_NAME.

SQL> CREATE TABLE E13 AS SELECT FIRST_NAME || DEPARTMENT_NAME EDNAME FROM EMPLOYEES	
	JOIN DEPARTMENTS USING (DEPARTMENT_ID);


14 CREATE TABLE E14 WHICH WILL HAVE DEPARTMENT_NAME AND HIGHEST SALARY PAID FOR EACH.

SQL> CREATE TABLE E14 AS SELECT DEPARTMENT_NAME, HIGHEST_SAL FROM DEPARTMENTS
	JOIN (SELECT DEPARTMENT_ID, MAX(SALARY) HIGHEST_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID) 
	USING (DEPARTMENT_ID);

15 CREATE TABLE E15 WHICH WILL STORE ALL COLUMNS OF EMPLOYEES TABLE BUT ONLY ROWS BETWEEN 12 AND 18.

SQL> CREATE TABLE E15 AS SELECT * FROM EMPLOYEES OFFSET 11 ROW FETCH NEXT 7 ROWS ONLY;
