1 DISPLAY ONLY THOSE DEPARTMENT_NAME IN WHICH NO EMPLOYEE IS HIRED
   SELECT DEPARTMENT_NAME
   FROM EMPLOYEES E
   RIGHT JOIN DEPARTMENTS D ON(E.DEPARTMENT_ID=D.DEPARTMENT_ID) WHERE E.DEPARTMENT_ID IS NULL;

  FROM EMPLOYEES E
  LEFT JOIN DEPARTMENTS D ON E.DEPARTMENT_ID=D.DEPARTMENT_ID WHERE E.DEPARTMENT_ID IS NULL;
