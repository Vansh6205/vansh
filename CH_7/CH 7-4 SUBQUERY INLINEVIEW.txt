1 DISPLAY FIRST_NAME , DEPARTMENT_ID , SALARY , HIGHEST SALARY PAID TO THE CORRESPONDING DEPARTMENT_ID
  SELECT FIRST_NAME,DEPARTMENT_ID,SALARY,HIGHEST_SALARY
  FROM EMPLOYEES 
  NATURAL JOIN (SELECT DEPARTMENT_ID,MAX(SALARY) HIGHEST_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

2 DISPLAY LAST_NAME , JOB_ID , SALARY , LOWEST SALARY PAID TO THE CORRESPONDING JOB_ID
  SELECT LAST_NAME,JOB_ID,SALARY,LOWEST_SALARY
  FROM EMPLOYEES 
  NATURAL JOIN (SELECT JOB_ID,MIN(SALARY) LOWEST_SALARY FROM EMPLOYEES GROUP BY JOB_ID); 

3 DISPLAY FIRST_NAME , MANAGER_ID , SALARY , AVERAGE SALARY PAID TO THE CORRESPONDING MANAGER_ID
  SELECT FIRST_NAME,MANAGER_ID,SALARY,AVERAGE_SALARY
  FROM EMPLOYEES 
  NATURAL JOIN (SELECT MANAGER_ID,AVG(SALARY) AVERAGE_SALARY FROM EMPLOYEES GROUP BY MANAGER_ID);

4 DISPLAY FIRST_NAME , HIRE YEAR , SALARY , HIRE YEAR WISE LOWEST SALARY , HIRE YEAR WISE HIGHEST SALARY
  SELECT FIRST_NAME,HIRE_YEAR,SALARY,HY_MIN_SAL,HY_MAX_SAL
  FROM (SELECT FIRST_NAME,TO_CHAR(HIRE_DATE,'RRRR') HIRE_YEAR,SALARY FROM EMPLOYEES)
  NATURAL JOIN (SELECT TO_CHAR(HIRE_DATE,'RRRR') HIRE_YEAR,MIN(SALARY) HY_MIN_SAL,MAX(SALARY) HY_MAX_SAL FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,'RRRR'));

5 DISPLAY FIRST_NAME , SALARY , DEPARTMENT_ID , DEPARTMENT WISE HIGHEST SALARY , JOB_ID , JOB WISE LOWEST SALARY
  SELECT FIRST_NAME,SALARY,DEPARTMENT_ID,D_HIGHEST_SALARY,JOB_ID,J_HIGHEST_SALARY
  FROM (SELECT DEPARTMENT_ID,MAX(SALARY) D_HIGHEST_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID) 
  NATURAL JOIN EMPLOYEES
  NATURAL JOIN (SELECT JOB_ID,MAX(SALARY) J_HIGHEST_SALARY FROM EMPLOYEES GROUP BY JOB_ID);

6 DISPLAY LAST_NAME , DEPARTMENT_ID , SALARY , LOWEST SALARY , HIGHEST SALARY FOR THE CORRESPONDING DEPARTMENT_ID
  SELECT LAST_NAME,DEPARTMENT_ID,SALARY,HIGHEST_SALARY,LOWEST_SALARY
  FROM (SELECT DEPARTMENT_ID,MAX(SALARY) HIGHEST_SALARY,MIN(SALARY) LOWEST_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID) 
  NATURAL JOIN EMPLOYEES;

7 DISPLAY FIRST_NAME , JOB_ID , SALARY , HIGHEST SALARY , DIFFERENCE (HIGHEST SALARY - SALARY) FOR THE CORRESPONDING JOB_ID
  SELECT FIRST_NAME,JOB_ID,SALARY,HIGHEST_SALARY,HIGHEST_SALARY-SALARY DIFFERENCE
  FROM EMPLOYEES 
  NATURAL JOIN (SELECT JOB_ID,MAX(SALARY) HIGHEST_SALARY FROM EMPLOYEES GROUP BY JOB_ID);

8 DISPLAY FIRST_NAME , LAST_NAME , DEPARTMENT_ID , JOB_ID , NO. OF EMPLOYEES WORKING FOR DEPARTMENT_ID , NO. OF EMPLOYEES WORKING FOR JOB_ID
  SELECT FIRST_NAME,LAST_NAME,DEPARTMENT_ID,JOB_ID,DEPT_COUNT,JOB_COUNT
  FROM (SELECT JOB_ID,COUNT(*) JOB_COUNT FROM EMPLOYEES GROUP BY JOB_ID) 
  NATURAL JOIN EMPLOYEES
  NATURAL JOIN (SELECT DEPARTMENT_ID,COUNT(*) DEPT_COUNT FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

9 DISPLAY FIRST_NAME , JOB_ID , SALARY , TOTAL SALARY PAID TO JOB_ID , %OF SALARY EMPLOYEE GET FOR THE CORRESPONDING JOB_ID
  SELECT FIRST_NAME,JOB_ID,SALARY,TOTAL_SALARY,(SALARY*100)/TOTAL_SALARY||'%' PERCENT
  FROM (SELECT JOB_ID,SUM(SALARY) TOTAL_SALARY FROM EMPLOYEES GROUP BY JOB_ID) 
  NATURAL JOIN EMPLOYEES;

10 DISPLAY FIRST_NAME , DEPARTMENT_ID , SALARY , LOWEST SALARY , DIFFERENCE IN % FOR THE CORRESPONDING DEPARTMENT_ID
   SELECT FIRST_NAME,DEPARTMENT_ID,SALARY,LOWEST_SALARY,(SALARY-LOWEST_SALARY)*100/LOWEST_SALARY||'%' PERCENT
   FROM (SELECT DEPARTMENT_ID,MIN(SALARY) LOWEST_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID) 
   NATURAL JOIN EMPLOYEES;

11 DISPLAY FIRST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE HIGHEST SALARY , JOB_ID , JOB WISE LOWEST SALARY
   SELECT FIRST_NAME,SALARY,DEPARTMENT_ID,DEPT_MAX_SALARY,JOB_ID,JOB_MAX_SALARY
   FROM (SELECT DEPARTMENT_ID,MAX(SALARY) DEPT_MAX_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID) 
   NATURAL JOIN EMPLOYEES
   NATURAL JOIN (SELECT JOB_ID,MAX(SALARY) JOB_MAX_SALARY FROM EMPLOYEES GROUP BY JOB_ID);

12 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE AVERAGE SALARY , MANAGER_ID , MANAGER WISE NO. OF EMPLOYEES
   SELECT LAST_NAME,SALARY,DEPARTMENT_ID,DEPT_AVG_SALARY,MANAGER_ID,MGR_COUNT_EMP
   FROM (SELECT DEPARTMENT_ID,AVG(SALARY) DEPT_AVG_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID) 
   NATURAL JOIN EMPLOYEES
   NATURAL JOIN (SELECT MANAGER_ID,COUNT(*) MGR_COUNT_EMP FROM EMPLOYEES GROUP BY MANAGER_ID);

13 DISPLAY LAST_NAME , SALARY , MANAGER_ID , MGR WISE AVERAGE SALARY , JOB_ID , JOB WISE NO. OF EMPLOYEES
   SELECT LAST_NAME,SALARY,MANAGER_ID,MGR_AVG_SALARY,JOB_ID,JOB_COUNT_EMP
   FROM (SELECT JOB_ID,COUNT(*) JOB_COUNT_EMP FROM EMPLOYEES GROUP BY JOB_ID) 
   NATURAL JOIN EMPLOYEES
   NATURAL JOIN (SELECT MANAGER_ID,AVG(SALARY) MGR_AVG_SALARY FROM EMPLOYEES GROUP BY MANAGER_ID);

14 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE LOWEST SALARY , JOB_ID , JOB WISE AVERAGE SALARY, MANAGER_ID , MANAGER WISE HIGHEST SALARY
   SELECT LAST_NAME,SALARY,DEPARTMENT_ID,DEPT_MIN_SALARY,JOB_ID,JOB_AVG_SALARY,MANAGER_ID,MGR_MAX_SALARY
   FROM (SELECT JOB_ID,AVG(SALARY) JOB_AVG_SALARY FROM EMPLOYEES GROUP BY JOB_ID) 
   NATURAL JOIN EMPLOYEES
   NATURAL JOIN (SELECT MANAGER_ID,MAX(SALARY) MGR_MAX_SALARY FROM EMPLOYEES GROUP BY MANAGER_ID)
   NATURAL JOIN ((SELECT DEPARTMENT_ID,MIN(SALARY) DEPT_MIN_SALARY FROM EMPLOYEES GROUP BY DEPARTMENT_ID));
   
15 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE HIGHEST SALARY , DEPARTMENT_NAME
   SELECT LAST_NAME,SALARY,DEPARTMENT_ID,DEPT_MAX_SAL,DEPARTMENT_NAME
   FROM EMPLOYEES 
   NATURAL JOIN (SELECT DEPARTMENT_ID,MAX(SALARY) DEPT_MAX_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
   NATURAL JOIN DEPARTMENTS;
16 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE AVERAGE SALARY , MANAGER_ID , MANAGER WISE NO. OF EMPLOYEES , DEPARTMENT_NAME , CITY
   SELECT LAST_NAME,SALARY,DEPARTMENT_ID,DEPT_AVG_SAL,MANAGER_ID,MGR_COUNT_EMP,DEPARTMENT_NAME,CITY
   FROM EMPLOYEES 
   NATURAL JOIN (SELECT DEPARTMENT_ID,AVG(SALARY) DEPT_AVG_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
   NATURAL JOIN (SELECT MANAGER_ID,COUNT(*) MGR_COUNT_EMP FROM EMPLOYEES GROUP BY MANAGER_ID)
   NATURAL JOIN DEPARTMENTS
   NATURAL JOIN LOCATIONS;
17 DISPLAY LAST_NAME , SALARY , MANAGER_ID , MGR WISE AVERAGE SALARY , JOB_ID , JOB WISE NO. OF EMPLOYEES , DEPARTMENT_NAME , CITY , COUNTRY_NAME
   SELECT LAST_NAME,SALARY,MANAGER_ID,MGR_AVG_SAL,JOB_ID,JOB_COUNT_EMP,DEPARTMENT_NAME,CITY,COUNTRY_NAME
   FROM EMPLOYEES 
   NATURAL JOIN (SELECT JOB_ID,COUNT(*) JOB_COUNT_EMP FROM EMPLOYEES GROUP BY JOB_ID)
   NATURAL JOIN (SELECT MANAGER_ID,AVG(SALARY) MGR_AVG_SAL FROM EMPLOYEES GROUP BY MANAGER_ID)
   NATURAL JOIN DEPARTMENTS
   NATURAL JOIN LOCATIONS
   NATURAL JOIN COUNTRIES;
18 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE LOWEST SALARY , JOB_ID , JOB WISE AVERAGE SALARY, MANAGER_ID , MANAGER WISE HIGHEST SALARY , DEPARTMENT_NAME , CITY , COUNTRY_NAME , REGION_NAME
   SELECT LAST_NAME,SALARY,DEPARTMENT_ID,DEPT_MIN_SAL,JOB_ID,JOB_AVG_SAL,MANAGER_ID,MGR_MAX_SAL,DEPARTMENT_NAME,CITY,COUNTRY_NAME,REGION_NAME
   FROM EMPLOYEES 
   NATURAL JOIN (SELECT JOB_ID,AVG(SALARY) JOB_AVG_SAL FROM EMPLOYEES GROUP BY JOB_ID)
   NATURAL JOIN (SELECT MANAGER_ID,MAX(SALARY) MGR_MAX_SAL FROM EMPLOYEES GROUP BY MANAGER_ID)
   NATURAL JOIN (SELECT DEPARTMENT_ID,MIN(SALARY) DEPT_MIN_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
   NATURAL JOIN DEPARTMENTS
   NATURAL JOIN LOCATIONS
   NATURAL JOIN COUNTRIES
   NATURAL JOIN REGIONS;
